691inline int GetOrientation(
692 Vector2 const& p1,
693 Vector2 const& p2,
694 Vector2 const& p3
695)
696{
697 //* https://www.geeksforgeeks.org/orientation-3-ordered-points/
698 float n{ ( p2.y - p1.y ) * ( p3.x - p2.x ) - ( p2.x - p1.x ) * ( p3.y - p2.y ) };
699 return n / abs( n );
700}

702inline bool CheckCollisionLines(
703 Vector2 const& a1,
704 Vector2 const& b1,
705 Vector2 const& c2,
706 Vector2 const& d2
707)
708{
709 //* https://www.geeksforgeeks.org/check-if-two-given-line-segments-intersect/
710 //* REQUIREMENT:
711 //* - Orientation of a1-b1-c2 and a1-b1-d2 differs
712 //* AND
713 //* - Orientation of c2-d2-a1 and c2-d2-b1 differs
714 //* CONCLUSION: No collision if
715 //* - one orientation is colinear (0) := touch but no intersection
716 //* OR
717 //* - one orientation pair is equal
718
719 int abc{ GetOrientation( a1, b1, c2 ) };
720 int abd{ GetOrientation( a1, b1, d2 ) };
721
722 int cda{ GetOrientation( c2, d2, a1 ) };
723 int cdb{ GetOrientation( c2, d2, b1 ) };
724
725 if ( //* Colinear orientations
726 !( abc * abd * cda * cdb )
727 //* No intersection
728 || ( abc == abd )
729 || ( cda == cdb )
730 )
731 {
732 return false;
733 }
734
735 return true;
736}
